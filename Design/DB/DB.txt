고객 기본 정보
CREATE TABLE customer(
  id VARCHAR(20) NOT NULL,
  pwd VARCHAR(30) NOT NULL,
  email VARCHAR(30) NOT NULL,
  grade CHAR(1) DEFAULT 'A',
  PRIMARY KEY(id)
) DEFAULT CHARACTER SET UTF8 COLLATE utf8_general_ci;

CREATE TABLE conform(
  email VARCHAR(30) NOT NULL,
  code CHAR(6) NOT NULL,
  PRIMARY KEY(email)
) DEFAULT CHARACTER SET UTF8 COLLATE utf8_general_ci;

게시판
CREATE TABLE board(
  pageid INT UNSIGNED NOT NULL AUTO_INCREMENT,
  title TEXT(40),
  content TEXT,
  author varchar(20) DEFAULT 'Anomyous',
  id VARCHAR(20),
  pwd VARCHAR(30),
  time DATETIME,
  view INT UNSIGNED,
  up INT UNSIGNED,
  down INT UNSIGNED,
  grade CHAR(1) DEFAULT 'N',
  PRIMARY KEY(pageid)
) DEFAULT CHARACTER SET UTF8 COLLATE utf8_general_ci;


댓글
CREATE TABLE reply(
  pageid INT,
  reid INT UNSIGNED,
  author VARCHAR(30) DEFAULT 'Anomyous',
  id VARCHAR(20),
  pwd VARCHAR(30),
  content TEXT(255),
  time DATETIME,
  up INT UNSIGNED,
  down INT UNSIGNED,
  FOREIGN KEY(pageid) REFERENCES board(pageid),
) DEFAULT CHARACTER SET UTF8 COLLATE utf8_general_ci;

대댓글
CREATE TABLE rereply(
  pageid INT,
  reid INT UNSIGNED,
  rereid INT UNSIGNED,
  author VARCHAR(30) DEFAULT 'Anomyous',
  id VARCHAR(20),
  pwd VARCHAR(30),
  content TEXT(255),
  time DATETIME,
  up INT UNSIGNED,
  down INT UNSIGNED,
  FOREIGN KEY(pageid) REFERENCES board(pageid),
  FOREIGN KEY(reid) REFERENCES reply(reid)
) DEFAULT CHARACTER SET UTF8 COLLATE utf8_general_ci;
