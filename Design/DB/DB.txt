CREATE TABLE customer(
  id VARCHAR(20) NOT NULL,
  pwd VARCHAR(30) NOT NULL,
  email VARCHAR(30) NOT NULL,
  grade CHAR(1) DEFAULT 'A',
  PRIMARY KEY(id)
) DEFAULT CHARACTER SET UTF8 COLLATE utf8_general_ci;

게시판
CREATE TABLE board(
  pageid INT UNSIGNED NOT NULL AUTO_INCREMENT,
  title TEXT(100),
  content TEXT,
  author varchar(20) DEFAULT 'Anomyous',
  id VARCHAR(20),
  pwd VARCHAR(30),
  time DATETIME,
  view INT UNSIGNED,
  up INT UNSIGNED,
  down INT UNSIGNED,
  grade CHAR(1) DEFAULT 'N',
  PRIMARY KEY(pageid)
) DEFAULT CHARACTER SET UTF8 COLLATE utf8_general_ci;


댓글
CREATE TABLE reply(
  pageid INT,
  reid INT UNSIGNED,
  author VARCHAR(30) DEFAULT 'Anomyous',
  id VARCHAR(20),
  pwd VARCHAR(30),
  content TEXT(300),
  time DATETIME,
  up INT UNSIGNED,
  down INT UNSIGNED,
  FOREIGN KEY(reid) REFERENCES board(pageid)
) DEFAULT CHARACTER SET UTF8 COLLATE utf8_general_ci;


대댓글
CREATE TABLE rereply(
  pageid INT,
  reid INT UNSIGNED,
  rereid INT UNSIGNED,
  author VARCHAR(30) DEFAULT 'Anomyous',
  id VARCHAR(20),
  pwd VARCHAR(30),
  content TEXT(300),
  time DATETIME,
  up INT UNSIGNED,
  down INT UNSIGNED,
  FOREIGN KEY(reid) REFERENCES board(pageid),
  FOREIGN KEY(rereid) REFERENCES reply(reid)
) DEFAULT CHARACTER SET UTF8 COLLATE utf8_general_ci;
